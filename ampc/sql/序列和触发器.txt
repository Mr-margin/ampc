------T_ADDRESS表 区域表----
CREATE SEQUENCE T_ADDRESS_SEQ  
INCREMENT BY 1   
START WITH 1      
NOMAXVALUE        
NOCYCLE               
CACHE 20;

CREATE OR REPLACE TRIGGER T_ADDRESS_REP before insert on T_ADDRESS
FOR EACH ROW
when (new.ADDRESS_ID is null)
begin
select T_ADDRESS_SEQ.nextval into :new.ADDRESS_ID from dual;
end;
